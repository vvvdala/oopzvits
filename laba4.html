Умова задачі

Завдання лабораторної роботи № 4 передбачають такі дії. Завдання 1 вимагає розробку базових і похідних класів з закритими полями та відкритими властивостями. Завдання 2 вимагає додавання до базового класу конструкторів з параметрами та без параметрів. Завдання 3 вимагає додавання до похідних класів конструкторів з параметрами. Завдання 4 передбачає розробку властивостей get, set для доступу до закритих полів класу. Завдання 5 передбачає розробку методів виведення на консоль значень полів класу. Завдання 6 передбачає розробку методів за сценарієм роботи програми. Завдання 7 є другою версією консольного застосунку, що реалізує ієрархію класів у вигляді інтерфейсу та похідних від інтерфейсу класів, що реалізують інтерфейс. Завдання 8 є третьою версією консольного застосунку, що реалізує ієрархію класів у вигляді абстрактного базового та похідних класів. Завдання 9 є четвертою версією консольного застосунку, що реалізує стандартні інтерфейси.

Виконувати це будемо за допомогою IComparable, IComparer, IEnumerable.

            Console.WriteLine("Радіус = " + centreRadius);

        public async void Radius()
        {
            string path = "circle.txt";
            centreRadius = Math.Sqrt((corX2 - corX) * (corX2 - corX) + (corY2 - corY) * (corY2 - corY));
            centreX = (corX + corX2 )/ 2;
            centreY = (corY + corY2) / 2;

            using (StreamWriter writer = new StreamWriter(path, true))
            {
                await writer.WriteLineAsync("Радіус: " + centreRadius);
                await writer.WriteLineAsync("X centre: " + centreX);
                await writer.WriteLineAsync("Y centre: " + centreY);

            }

        }
        public async void Area()
        {
            double area = centreRadius * centreRadius * Math.PI;
            Console.WriteLine("Площа = " + area);

            string path = "circle.txt";
            using (StreamWriter writer = new StreamWriter(path, true))
            {
                await writer.WriteLineAsync("Площа: " + area);
                

            }
        }

        public async void RadiusTriangle()
        {
            Console.WriteLine("Введіть значення сторон трикутника");
            Console.Write("1 cторона:");
            double length1=double.Parse(Console.ReadLine());
            Console.WriteLine();
            Console.Write("2 cторона:");
            double length2 = double.Parse(Console.ReadLine());
            Console.WriteLine();
            Console.Write("3 cторона:");
            double length3 = double.Parse(Console.ReadLine());
            Console.WriteLine();
            double halfP;
            halfP = (length1 + length2 + length3) / 2;
            double areaTriangle=Math.Sqrt(halfP*(halfP-length1)*(halfP - length2)*(halfP - length3));
            double inRadius = areaTriangle / halfP;
            double outRadius = (length1 * length2 * length3) / (4 * areaTriangle);

            Console.WriteLine("R = "+outRadius);
            Console.WriteLine("r = " + inRadius);


            string path = "circle.txt";
            using (StreamWriter writer = new StreamWriter(path, true))
            {
                await writer.WriteLineAsync("R: " + outRadius);
                await writer.WriteLineAsync("r: " + inRadius);

            }
        }

        public async void RadiusSquare()
        {
            Console.Write("Cторона квадрата:");
            double a = double.Parse(Console.ReadLine());
            Console.WriteLine();
            double inRadius = a / 2;
            double outRadius = a / Math.Sqrt(2);

            Console.WriteLine("R = " + outRadius);
            Console.WriteLine("r = " + inRadius);

            string path = "circle.txt";
            using (StreamWriter writer = new StreamWriter(path, true))
            {
                await writer.WriteLineAsync("R: " + outRadius);
                await writer.WriteLineAsync("r: " + inRadius);

            }
        }

        public async void RadiusRectangle()
        {
            Console.Write("Cторона 1:");
            double a = double.Parse(Console.ReadLine());
            Console.WriteLine();

            Console.Write("Cторона 2:");
            double b = double.Parse(Console.ReadLine());
            Console.WriteLine();

            if (a <= b)
            {
                Console.WriteLine("Спробуйте ще раз!:");
                return;
            }

            double outRadius = (Math.Sqrt(a * a - b * b)) / 2;

            Console.WriteLine("R = " + outRadius);

            string path = "circle.txt";
            using (StreamWriter writer = new StreamWriter(path, true))
            {
                await writer.WriteLineAsync("R: " + outRadius);

            }
        }
    }

Наслідування, абстрактні класи та інтерфейс - це основні концепції об'єктно-орієнтованого програмування, що дозволяють писати більш структурований код, який можно розширяти та використовувати повторно. Наслідування дозволяє створювати нові класи, що успадковують властивості та методи від батьківського класу, тим самим зменшуючи кількість дублюючого коду та спрощуючи розробку застосунку. Абстрактні класи є частково реалізованим класом, який містить абстрактні методи, що повинні бути реалізовані в дочірніх класах. Це дозволяє вимагати від класів-спадкоємців реалізації певного набору методів та забезпечує більш гнучкий та безпечний процес розробки програм. Інтерфейс - це набір методів, що повинні бути реалізовані в класі, який реалізує цей інтерфейс. Інтерфейси дозволяють забезпечити наявність певного набору методів у класі, незалежно від його реалізації, тим самим дозволяючи взаємодіяти з класом безпосередньо через інтерфейс, а не через конкретну реалізацію. Також він дає змогу успадковувати багато ''класів''.
